这道题里说的bug，复现方法就是如下客户端代码，发送一半的数据，不带\n，那么服务器端就会因为读不到\n而一直阻塞，从而实现服务器端的拒绝服务bug
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>

int main(){
    //创建套接字
    int connfd = socket(AF_INET, SOCK_STREAM, 0);
    struct linger lr;
    lr.l_onoff=1;
    lr.l_linger=0;
    setsockopt(connfd, SOL_SOCKET, SO_LINGER, &lr, sizeof(struct linger));
    
    //向服务器（特定的IP和端口）发起请求
    struct sockaddr_in serv_addr;
    memset(&serv_addr, 0, sizeof(serv_addr));  //每个字节都用0填充
    serv_addr.sin_family = AF_INET;  //使用IPv4地址
    serv_addr.sin_addr.s_addr = inet_addr("10.156.64.44");  //具体的IP地址
    serv_addr.sin_port = htons(8000);  //端口
    connect(connfd, (struct sockaddr*)&serv_addr, sizeof(serv_addr));
    
    //读取服务器传回的数据
    char buffer[400];
    sprintf(buffer, "Content-type: text");
    printf("send buffer size:%ld \n", strlen(buffer));
    write(connfd, buffer, strlen(buffer));
    sleep(3);
    size_t n;
    if ((n = read(connfd, buffer, 400)) > 0) {
        write(STDOUT_FILENO, buffer, n);
    }
    sprintf(buffer, "/html\r\n");
    printf("send buffer size:%ld \n", strlen(buffer));
    write(connfd, buffer, strlen(buffer));
    sleep(3);
    if ((n = read(connfd, buffer, 400)) > 0) {
        write(STDOUT_FILENO, buffer, n);
    }
    //关闭套接字
//    close(connfd);
    pause();
    
    return 0;
}

```
一开始的想法是做成一个非阻塞的IO，但是发现客户端发送数据后会输出nonblock error ingored!，然后一直卡住。推断是因为read每次读n个字节（本例n=10），阻塞IO的场景下不足n个字节就会阻塞，只有遇到EOF、对端关闭才返回。非阻塞情况下不足n个字节会直接返回，并设置errno，但是接下来select因为没有检测到有新数据到会直接阻塞
```c
void check_clients(pool *p)
{
    int i, connfd, n;
    char buf[MAXLINE];
    rio_t rio;

    for (i = 0; (i <= p->maxi) && (p->nready > 0); i++) {
        connfd = p->clientfd[i];
        rio = p->clientrio[i];

        /* If the descriptor is ready, echo a text line from it */
        if ((connfd > 0) && (FD_ISSET(connfd, &p->ready_set))) {
            int flags;
            // 使用非阻塞IO
            if ((flags = fcntl(connfd, F_GETFL, 0)) < 0) {
                app_error("fcntl error: fail to get flags");
                exit(0);
            }
            flags |= O_NONBLOCK;
            if(fcntl(connfd, F_SETFL, flags) < 0) {
                app_error("fcntl error: fail to set flags");
                exit(0);
            }
            p->nready--;
            if ((n = Rio_readnb(&rio, buf, 10)) != 0) {
                if (n < 0) {
                    printf("nonblock error ingored!\n");
                    continue;
                }
                byte_cnt += n;
                printf("Server received %d (%d total) bytes on fd %d\n",
                       n, byte_cnt, connfd);
                Rio_writen(connfd, buf, n);
            }

            /* EOF detected, remove descriptor from pool */
            else {
                Close(connfd);
                FD_CLR(connfd, &p->read_set);
                p->clientfd[i] = -1;
            }
        }
    }
}
```

改进后使用了ioctl包，先获取套接字内核缓冲区中有多少字节数据，然后读取出来，修复了拒绝服务的bug
```c
void check_clients(pool *p)
{
    int i, connfd, n;
    char buf[MAXLINE];
    rio_t rio;

    for (i = 0; (i <= p->maxi) && (p->nready > 0); i++) {
        connfd = p->clientfd[i];
        rio = p->clientrio[i];

        /* If the descriptor is ready, echo a text line from it */
        if ((connfd > 0) && (FD_ISSET(connfd, &p->ready_set))) {
            size_t avail =0;
            ioctl(connfd, FIONREAD, &avail);
            printf("ioctl avail bytes = %d\n", avail);
            p->nready--;
            if ((n = Rio_readnb(&rio, buf, avail)) != 0) {
                byte_cnt += n;
                printf("Server received %d (%d total) bytes on fd %d\n",
                       n, byte_cnt, connfd);
                Rio_writen(connfd, buf, n);
            }

            /* EOF detected, remove descriptor from pool */
            else {
                Close(connfd);
                FD_CLR(connfd, &p->read_set);
                p->clientfd[i] = -1;
            }
        }
    }
}
```
