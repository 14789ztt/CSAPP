#include <stdio.h>
void bubble_b(long *data, long count) {
    long i,last;
    for(last = count -1; last > 0; last--) {
        for(i=0; i < last; i++) 
            if (*(data+8*(i+1)) < *(data+8*i)) {
                long t = *(data+8*(i+1));
                *(data+8*(i+1)) = *(data+8*i);
                *(data+8*i) = t;
            }
    }
}
