/* $begin eval */
/* eval - Evaluate a command line */
void eval(char *cmdline)
{
    char *argv[MAXARGS]; /* Argument list execve() */
    char buf[MAXLINE];   /* Holds modified command line */
    int bg;              /* Should the job run in bg or fg? */
    pid_t pid;           /* Process id */
    sigset_t mask_all, mask_one, prev;
    sigfillset(&mask_all);
    sigemptyset(&mask_one);
    sigaddset(&mask_one, SIGCHLD);
    strcpy(buf, cmdline);
    bg = parseline(buf, argv);
    if (argv[0] == NULL)
        return;   /* Ignore empty lines */
    
    if (!builtin_command(argv)) {
        sigprocmask(SIG_BLOCK, &mask_one, &prev);//阻塞SIGCHID，防止在add_job之前就结束delete_job
        if ((pid = fork()) == 0) {   /* Child runs user job */
            sigprocmask(SIG_SETMASK, &prev, NULL);//复制了父进程的block列表，所以也需要恢复
            if (execve(argv[0], argv, environ) < 0) {
                printf("%s: Command not found.\n", argv[0]);
                exit(0);
            }
        }
        
        sigprocmask(SIG_BLOCK, &mask_all, NULL);
        add_job(pid, cmdline);
        sigprocmask(SIG_SETMASK, &prev, NULL);
        

        /* Parent waits for foreground job to terminate */
        if (!bg) {
            // 这里应该改成suspend，不然sigchld信号会在chld_handler里处理完，这里会报No child processes
            // int status;
            // if (waitpid(pid, &status, 0) < 0)
            //    unix_error("waitfg: waitpid error");
            fg_gpid = pid;
            while(fg_gpid) {
                sigsuspend(&prev);
            }
        }
        else
            printf("%d %s", pid, cmdline);
    }
    return;
}

/* If first arg is a builtin command, run it and return true */
int builtin_command(char **argv)
{
    if (!strcmp(argv[0], "quit"))   /* quit command */
        exit(0);
    if (!strcmp(argv[0], "&"))      /* Ignore singleton & */
        return 1;
    if (!strcmp(argv[0], "jobs")) { /* list all jobs */
        list_jobs();
        return 1;
    }
    if (!strcmp(argv[0], "bg")) {   /* run job in background */
        run_background(argv[1]);
        return 1;
    }
    if (!strcmp(argv[0], "fg")) {   /* run job in frontground */
        run_frontground(argv[1]);
        return 1;
    }
    return 0;                     /* Not a builtin command */
}

/* $begin parseline */
/* parseline - Parse the command line and build the argv array */
int parseline(char *buf, char **argv)
{
    char *delim;         /* Points to first space delimiter */
    int argc;            /* Number of args */
    int bg;              /* Background job? */

    buf[strlen(buf)-1] = ' ';  /* Replace trailing '\n' with space */
    while (*buf && (*buf == ' ')) /* Ignore leading spaces */
        buf++;

    /* Build the argv list */
    argc = 0;
    while ((delim = strchr(buf, ' '))) {
        argv[argc++] = buf;
        *delim = '\0';
        buf = delim + 1;
        while (*buf && (*buf == ' ')) /* Ignore spaces */
                buf++;
    }
    argv[argc] = NULL;
    
    if (argc == 0)  /* Ignore blank line */
        return 1;

    /* Should the job run in the background? */
    if ((bg = (*argv[argc-1] == '&')) != 0)
        argv[--argc] = NULL;

    return bg;
}
/* $end parseline */

void sigchld_handler(int s)
{printf("=====fg %d \n", fg_gpid);
    int olderrno = errno;
    int status;
    pid_t pid;
    // 循环回收所有停止或者结束的子进程
    // 设置WNOHANG | WUNTRACED | WCONTINUED是为了在暂停/重启进程时也能立即返回，不会卡住
    while((pid = waitpid(-1, &status, WNOHANG | WUNTRACED | WCONTINUED)) > 0) {
        if (WIFEXITED(status)) {
            // normal exit
            printf("Job %d normal exit\n",pid);
            delete_job(pid);
        }
        if (WIFSIGNALED(status)) {
            if (WTERMSIG(status) == SIGINT) {
                // User types Ctrl+C
                printf("Job %d terminated by signal: Interrupt\n", pid);
                delete_job(pid);
            }
            
            if (WTERMSIG(status) == SIGTERM) {
                // User types bin/kill pid
                printf("Job %d terminated by signal: Terminated\n", pid);
                delete_job(pid);
            }
        }
        if (WIFSTOPPED(status)) {
            printf("Job %d stopped by signal: Stopped\n", pid);
            update_job_status(pid, "stopped");
        }
        if(fg_gpid == pid) {
            // 如果退出的进程是前台进程，则fg_gpid置为0使得sigsuspend循环能退出
            fg_gpid = 0;
        }
    }
    printf("=====waitpid result: %d \n", pid);
    if (errno != ECHILD){
        printf("waitpid error\n");
    }
    errno = olderrno;
}

void sigint_handler(int s)
{
    if(fg_gpid != 0) {
        // 只处理前台进程的Ctrl+C
        printf("kill -SIGINT %d \n", fg_gpid);
        kill(fg_gpid, SIGINT);
    }
}

void sigtstp_handler(int s){
    printf("sigtstp_handler fg_gpid = %d\n", fg_gpid);
    if(fg_gpid != 0) {
        // 只处理前台进程的Ctrl+Z
        printf("kill -SIGTSTP %d \n", fg_gpid);
        kill(fg_gpid, SIGTSTP);
    }
}

void add_job(pid_t pid, char *cmdline)
{
    Job *job = (Job *) malloc(sizeof(Job));
    job->jid = seq++;
    job->pid = pid;
    strcpy(job->status, "running");
    strcpy(job->cmdline, cmdline);
    job->next = head_list->next;
    head_list->next = job;
    printf("add_job : [%d] %d %s %s\n", job->jid, job->pid, job->status, job->cmdline);
}

Job* get_job(pid_t pid)
{
    Job *p = head_list->next;
    while (p != NULL)
    {
        if (p->pid == pid) {
            printf("get job : [%d] %d %s %s\n", p->jid, p->pid, p->status, p->cmdline);
            return p;
        }
        
        p = p->next;
    }
    return NULL;
}

void delete_job(pid_t pid)
{
    Job *p = head_list;
    while (p->next != NULL)
    {
        Job* tmp = p->next;
        if (tmp->pid == pid) {
            printf("delete job : [%d] %d %s\n", tmp->jid, tmp->pid, tmp->cmdline);
            p->next = tmp->next;
            free(tmp);
            break;
        }
        
        p = p->next;
    }
}

void update_job_status(pid_t pid, char* status)
{
    Job *p = get_job(pid);
    if (p == NULL) {
        printf("job not found!\n");
        return;
    }
    strcpy(p->status, status);
}

{
    Job *p = head_list->next;
    while (p != NULL)
    {
       printf("[%d] %d %s %s\n", p->jid, p->pid, p->status, p->cmdline);
       p = p->next;
    }
    
}

void run_frontground(char *argv)
{
    pid_t pid = atoi(argv);
    Job *job = get_job(pid);
    if (job == NULL) {
        printf("%d: No such process\n", pid);
        return;
    }
    // kill旧进程
    kill(pid, SIGCONT);
    sigset_t prev;
    fg_gpid = pid;
    while(fg_gpid) {
        sigsuspend(&prev);
    }
}

void run_background(char *argv)
{
    pid_t pid = atoi(argv);
    Job *job = get_job(pid);
    if (job == NULL) {
        printf("%d: No such process\n", pid);
        return;
    }
    // 重启进程
    kill(pid, SIGCONT);
    update_job_status(pid, "running");
}
